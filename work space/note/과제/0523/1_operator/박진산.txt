package ex01.operatir;

public class Problem01 {
	public static void main(String[] args) {
		int pay = 567890;
		
		System.out.println("만원: " + pay / 10000 + "장");
		pay = pay % 10000;
		System.out.println("천원: " + pay / 1000 + "장");
		pay = pay % 1000;
		System.out.println("백원: " + pay / 100 + "원");
		pay = pay % 100;
		System.out.println("십원: " + pay / 10 + "원");
		
	} // end main
} // end class
