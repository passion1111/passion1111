package ex05_break_continue;

//과제] (-1)+2+(-3)+4+(-5)....10=?
public class day05_homework2 {
	public static void main(String[] args) {
		int temp = 0;
		int sum = -1;
		for (int i = 1; i != 11; i++) {
			if (i % 2 != 0)
				temp += i * sum;
			if (i % 2 == 0)
				temp += i;
					

		}
		System.out.println("총합은 =" + temp);

	}

}
