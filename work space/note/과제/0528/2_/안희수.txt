public class Quiz_two {
		public static void main(String[] args) {
			
			int num1 = 0;
			int num2 = 0;
			int result = 0;
			
			for(int i = 1; i <= 10; i++) {
				if(i % 2 == 0) {
					num1 += i;
				}else {
					num2 -= i;
				}
				result = num1+num2;
			}
			System.out.print(result);
		}
}
