package HW02;

public class HW_02 {

	public static void main(String[] args) {
		int sum_even = 0, sum_odd = 0;
		
		for (int i = 1; i <= 10; i++) {
			if(i % 2 == 0) sum_even += i; 
			else sum_odd += -i;
		}
		System.out.println(" (-1) + 2 + (-3) + 4 + (-5) + ... + (-9) + 10 = " + (sum_even + sum_odd));
	
	}

}
